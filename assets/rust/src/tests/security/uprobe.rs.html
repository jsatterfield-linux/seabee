3/// Module to check the security of uprobe protections
4///
5/// This requires that the `tests` crate has been built
6/// which contains the two BPF userspace binaries used for testing
7use anyhow::Context;
8use libtest_mimic::{Failed, Trial};
9
10use crate::{command::TestCommandBuilder, create_test};
11
12/// Tests that an arbitrary program using `bpf_write_user` is denied from loading
13fn security_uprobe_deny_write_user() -> Result<(), Failed> {
14    let exe_path = std::env::current_exe()?;
15    let target_folder = exe_path
16        .parent()
17        .context("Expected to be running inside of the target folder")?;
18    let binary_path = target_folder.join("bpf_write_user");
19    let binary_path_str = binary_path.to_str().context(format!(
20        "{:#?} could not be converted to string",
21        binary_path
22    ))?;
23    TestCommandBuilder::default()
24        .program(binary_path_str)
25        .expected_rc(101)
26        .build()?
27        .test()
28}
29
30pub fn tests() -> Vec<Trial> {
31    vec![create_test!(security_uprobe_deny_write_user)]
32}
