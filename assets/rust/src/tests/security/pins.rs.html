3/// Module to check the security of pin protections
4///
5/// # Intentionally unincluded Tests
6///
7/// - BPF_LINK_DETACH on our pinned programs returns -EOPNOTSUPP with bpftool.
8///
9/// # Missing Tests
10///
11/// - Is `BPF_LINK_UPDATE` dangerous to our links?
12/// - prevent `BPF_OBJ_GET` on pins, generally prevent interaction with
13///   our pinned programs through the BPF filesystem
14use std::path::Path;
15
16use libtest_mimic::{Failed, Trial};
17
18use crate::{command::TestCommandBuilder, create_test, suite::TestSuite};
19
20use super::SeaBeeSecurityTestSuite;
21
22/// Tests that protected BPF pins cannot be deleted
23fn security_pin_deny_unlink() -> Result<(), Failed> {
24    let state = SeaBeeSecurityTestSuite::get_system_state()?;
25
26    let dir = Path::new(&state.rust_state.pins.dir);
27    for path in state.rust_state.pins.pins.keys() {
28        let full_path = dir.join(Path::new(&path));
29        TestCommandBuilder::default()
30            .program("rm")
31            .args(&[&full_path.to_string_lossy()])
32            .expected_rc(1)
33            .expected_stderr("Operation not permitted")
34            .build()?
35            .test()?;
36    }
37
38    Ok(())
39}
40
41pub fn tests() -> Vec<Trial> {
42    vec![create_test!(security_pin_deny_unlink)]
43}
