4/// Module to check the security of file protections
5///
6/// IMPORTANT: if these tests fail, you may need to delete
7/// the `target` folder and recompile since they may corrupt
8/// the binary and configs used to do testing.
9use libtest_mimic::{Failed, Trial};
10
11use crate::{command::TestCommandBuilder, create_test, suite::TestSuite};
12
13use super::SeaBeeSecurityTestSuite;
14
15/// Attempts to remove a file testing that permission is denied
16fn try_unlink(path: &str) -> Result<(), Failed> {
17    TestCommandBuilder::default()
18        .program("rm")
19        .args(&[path])
20        .expected_rc(1)
21        .expected_stderr("Operation not permitted")
22        .build()?
23        .test()
24}
25
26/// Attempt to write to a file testing that permission is denied
27fn try_write(path: &str, exe: bool) -> Result<(), Failed> {
28    if exe {
29        TestCommandBuilder::default()
30            .program("bash")
31            .args(&["-c", &format!("echo test >> {path}")])
32            .expected_stderr("Text file busy")
33            .expected_rc(1)
34            .build()?
35            .test()
36    } else {
37        TestCommandBuilder::default()
38            .program("bash")
39            .args(&["-c", &format!("echo test >> {path}")])
40            .expected_stderr("Operation not permitted")
41            .expected_rc(1)
42            .build()?
43            .test()
44    }
45}
46
47/// Tests that protected files cannot be deleted
48fn security_file_deny_unlink() -> Result<(), Failed> {
49    let config = SeaBeeSecurityTestSuite::get_custom_state()?;
50
51    for file in &config.policy_file.files {
52        try_unlink(file)?
53    }
54
55    Ok(())
56}
57
58/// Tests that protected files cannot be written to
59fn security_file_deny_write() -> Result<(), Failed> {
60    let config = SeaBeeSecurityTestSuite::get_custom_state()?;
61
62    for file in &config.policy_file.files {
63        try_write(file, PathBuf::from(file) == std::env::current_exe()?)?
64    }
65
66    Ok(())
67}
68
69pub fn tests() -> Vec<Trial> {
70    vec![
71        create_test!(security_file_deny_unlink),
72        create_test!(security_file_deny_write),
73    ]
74}
