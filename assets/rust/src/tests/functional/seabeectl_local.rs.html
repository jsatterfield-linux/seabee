3/// This file tests the local commands of seabeectl including "alg", "sign", and "verify"
4/// sign will use keys besides ECDSA or RSA since there is no reason to restrict it.
5/// However, verify mirrors the seabeectl verify process and will not allow
6/// key types besides ECDSA or RSA
7use std::fs;
8
9use crate::{command::TestCommandBuilder, create_test};
10use libtest_mimic;
11use seabee::crypto;
12use strum::IntoEnumIterator;
13
14const POLICY: &str = "policies/test_policy.yaml";
15const RSA_PRIV: &str = "crypto/keys/rsa-private.pem";
16const RSA_PUB: &str = "crypto/keys/rsa-public.pem";
17const EC_PRIV: &str = "crypto/keys/ecdsa-private.pem";
18const EC_PUB: &str = "crypto/keys/ecdsa-public.pem";
19const SHA3: &str = "sha3-256";
20
21// invalid keys
22const FAKE_KEY: &str = "crypto/keys/invalid/fakekey";
23const DSA_PRIV: &str = "crypto/keys/invalid/dsa-private.pem";
24const DSA_PUB: &str = "crypto/keys/invalid/dsa-public.pem";
25const DNE: &str = "/some/nonexistant/file.abcdz";
26
27fn seabeectl_sign(
28    t: &str,
29    k: &str,
30    o: &str,
31    d: &str,
32    rc: i32,
33    stdout: &str,
34    stderr: &str,
35) -> Result<(), libtest_mimic::Failed> {
36    TestCommandBuilder::default()
37        .program("seabeectl")
38        .args(&["sign", "-t", t, "-k", k, "--nopass", "-d", d, "-o", o])
39        .expected_rc(rc)
40        .expected_stdout(stdout)
41        .expected_stderr(stderr)
42        .build()?
43        .test()
44}

65fn seabeectl_sign_verify_default() -> Result<(), libtest_mimic::Failed> {
66    // cannot have generic path due to parallelism
67    let sig = "default_sig_test.sign";
68    TestCommandBuilder::default()
69        .program("seabeectl")
70        .args(&["sign", "-t", POLICY, "-k", EC_PRIV, "--nopass", "-o", sig])
71        .expected_rc(0)
72        .expected_stdout("Success")
73        .build()?
74        .test()?;
75    TestCommandBuilder::default()
76        .program("seabeectl")
77        .args(&["verify", "-t", POLICY, "-k", EC_PUB, "-s", sig])
78        .expected_rc(0)
79        .expected_stdout("Verified OK")
80        .build()?
81        .test()?;
82    fs::remove_file(sig)?;
83    Ok(())
84}

97fn seabeectl_local_invalid_digest() -> Result<(), libtest_mimic::Failed> {
98    let sig = "invalid_sig"; // will not be created
99    seabeectl_sign(POLICY, EC_PRIV, sig, "md5", 2, "", "invalid value")?;
100    seabeectl_verify(POLICY, EC_PUB, sig, "md5", 2, "", "invalid value")?;
101    Ok(())
102}

104fn seabeectl_local_invalid_key_ext() -> Result<(), libtest_mimic::Failed> {
105    let sig = "invalid_key_ext.sign";
106    seabeectl_sign(POLICY, FAKE_KEY, sig, SHA3, 1, "", "expected extension")?;
107    // create sig path
108    seabeectl_sign(POLICY, EC_PRIV, sig, SHA3, 0, "Success", "")?;
109    seabeectl_verify(POLICY, FAKE_KEY, sig, SHA3, 1, "", "expected extension")?;
110    fs::remove_file(sig)?;
111    Ok(())
112}

114fn seabeectl_local_file_not_found() -> Result<(), libtest_mimic::Failed> {
115    let sig = "invalid_sig"; // will not be created
116    seabeectl_sign(
117        DNE,
118        RSA_PRIV,
119        sig,
120        SHA3,
121        1,
122        "",
123        "target path does not exist",
124    )?;
125    seabeectl_sign(POLICY, DNE, sig, SHA3, 1, "", "key path does not exist")?;
126    seabeectl_verify(DNE, RSA_PUB, sig, SHA3, 1, "", "target path does not exist")?;
127    seabeectl_verify(POLICY, DNE, sig, SHA3, 1, "", "key path does not exist")?;
128    seabeectl_verify(
129        POLICY,
130        RSA_PUB,
131        DNE,
132        SHA3,
133        1,
134        "",
135        "signature path does not exist",
136    )
137}

147fn seabeectl_sign_verify_ecdsa() -> Result<(), libtest_mimic::Failed> {
148    // cannot have generic path due to parallelism
149    let sig = "sign_verify_ecdsa.sign";
150    for digest in crypto::SeaBeeDigest::iter() {
151        let digest = digest.to_kebab_case();
152        seabeectl_sign(POLICY, EC_PRIV, sig, &digest, 0, "Success", "")?;
153        seabeectl_verify(POLICY, EC_PUB, sig, &digest, 0, "Verified OK", "")?;
154    }
155    fs::remove_file(sig)?;
156    Ok(())
157}
