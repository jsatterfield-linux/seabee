30const TEST_TOOL_PIN: &str = "/sys/fs/bpf/test_tool_pin";
31const TEST_PROG_NAME: &str = "test_seabee";

33pub fn start_test_tool() -> Result<Child> {
34    // add key
35    Command::new("seabeectl")
36        .args(["add-key", "-t", &utils::str_to_abs_path(RSA_PUB)?])
37        .stdout(std::process::Stdio::null())
38        .status()?;
39
40    // add policy
41    Command::new("seabeectl")
42        .args([
43            "update",
44            "-t",
45            &utils::str_to_abs_path(test_tool_config::TEST_TOOL_POLICY)?,
46            "-s",
47            &utils::str_to_abs_path(test_tool_config::TEST_TOOL_POLICY_SIG)?,
48        ])
49        .stdout(std::process::Stdio::null())
50        .status()?;
51
52    // start process
53    let child = Command::new(test_tool_config::TEST_TOOL_BIN)
54        .stdout(std::process::Stdio::null())
55        .spawn()?;
56
57    // wait up to 10 seconds for eBPF to load
58    for _timeout in 1..10 {
59        if std::path::Path::new(TEST_TOOL_PIN).exists() {
60            break;
61        }
62        std::thread::sleep(std::time::Duration::from_secs(1))
63    }
64    Ok(child)
65}
66
67pub fn stop_test_tool(child: Child) -> Result<()> {
68    // send Ctrl+C to process
69    let res = unsafe { libc::kill(child.id() as i32, libc::SIGINT) };
70    if res != 0 {
71        return Err(anyhow!("Failed to send SIGINT to test tool"));
72    }
73
74    // cleanup seabee
75    Command::new("seabeectl")
76        .args(["clean", "policy"])
77        .stdout(std::process::Stdio::null())
78        .status()?;
79    Command::new("seabeectl")
80        .args(["clean", "keys"])
81        .stdout(std::process::Stdio::null())
82        .status()?;
83
84    Ok(())
85}

98#[derive(Debug, Deserialize)]
99struct ProgInfo {
100    map_ids: Vec<u32>,
101}
102
103fn deny_map_access() -> Result<(), Failed> {
104    let output = Command::new("bpftool")
105        .args(["prog", "show", "name", TEST_PROG_NAME, "-p"])
106        .output()?;
107    let json_out: ProgInfo = serde_json::from_str(&String::from_utf8(output.stdout)?)?;
108
109    let mut found_map = false;
110    for id in json_out.map_ids {
111        found_map = true;
112        TestCommandBuilder::default()
113            .program("bpftool")
114            .args(&["map", "dump", "id", &id.to_string()])
115            .expected_rc(255)
116            .expected_stderr("Operation not permitted")
117            .build()?
118            .test()?
119    }
120
121    if !found_map {
122        return Err(anyhow!("Failed to find any maps for test bpf program").into());
123    }
124    Ok(())
125}
