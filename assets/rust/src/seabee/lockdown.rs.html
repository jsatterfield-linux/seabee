2use std::collections::HashSet;
3
4use anyhow::Result;
5use seccompiler::{BpfProgram, SeccompAction, SeccompFilter};
6
7use crate::config::Config;
8
9pub fn lockdown_seabee(config: &Config, prog_filter: &mut HashSet<String>) {
10    // if forking is allowed or a seccomp filter is created
11    // then do not load 'seabee_task_alloc'
12    if config.test || add_seccomp_filter().is_ok() {
13        prog_filter.insert(String::from("seabee_task_alloc"));
14    }
15}
16
17/// Attempts to add a syscall filter using seccomp to prevent any syscall
18/// that could copy the BPF file descriptors from running or change the
19/// execution of the userspace.
20///
21/// In addition, we prevent the `prctl` and `seccomp` syscalls to prevent
22/// further manipulation to the userspace
23///
24/// See `man 2 seccomp` for more information
25fn add_seccomp_filter() -> Result<()> {
26    let filter = SeccompFilter::new(
27        vec![
28            (libc::SYS_clone, vec![]),
29            (libc::SYS_clone3, vec![]),
30            (libc::SYS_execve, vec![]),
31            (libc::SYS_execveat, vec![]),
32            (libc::SYS_fork, vec![]),
33            (libc::SYS_prctl, vec![]),
34            (libc::SYS_seccomp, vec![]),
35            (libc::SYS_vfork, vec![]),
36        ]
37        .into_iter()
38        .collect(),
39        SeccompAction::Allow,
40        SeccompAction::Errno(libc::EPERM as u32),
41        std::env::consts::ARCH.try_into()?,
42    )?;
43    let seccomp_prog: BpfProgram = filter.try_into()?;
44    seccompiler::apply_filter_all_threads(&seccomp_prog)?;
45    Ok(())
46}
