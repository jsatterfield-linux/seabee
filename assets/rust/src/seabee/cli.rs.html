12/// The default arguments
13impl Default for Args {
14    fn default() -> Self {
15        Self {
16            // Do not specify a config file
17            config: None,
18            // Log Level Info
19            log_level: Some(LogLevelArg::info),
20            // Enable map protection
21            map_modification: Some(SecurityLevel::blocked),
22            // Enable pin protection
23            pin_modification: Some(SecurityLevel::blocked),
24            // sigint is not allowed
25            sigint: Some(SecurityLevel::blocked),
26            // kernel modules are allowed
27            kmod: Some(SecurityLevel::audit),
28            // daemon modification is not allowed
29            daemon_modification: Some(SecurityLevel::blocked),
30            // ptrace is not allowed
31            ptrace: Some(SecurityLevel::blocked),
32            // nothing is excluded
33            exclude: Vec::new(),
34            // digital signature verification is enabled
35            verify_policy: Some(true),
36            // by default anyone can add a key to SeaBee
37            verify_keys: Some(false),
38        }
39    }
40}

42/// Get config info from CLI arguments
43pub fn args_from_cli() -> Result<Args> {
44    let cli_args = Args::parse();
45    // Get args from a path provided on the command line
46    if cli_args.config.is_some() {
47        let mut file_args = args_from_file(cli_args.config.as_ref().unwrap())?;
48        file_args.apply(cli_args);
49        return Ok(file_args);
50    }
51    Ok(cli_args)
52}

54/// Get config info from a config file
55///
56/// <div class="warning">This funciton does not check if the file exists</div>
57pub fn args_from_file(config_path: &str) -> Result<Args> {
58    let config_str = std::fs::read_to_string(config_path)
59        .map_err(|e| anyhow!("Failed to read file: {config_path}\n{e}"))?;
60    let args = serde_yaml::from_str(&config_str)
61        .map_err(|e| anyhow!("Failed to parse file into config: {config_path}\n{e}"))?;
62
63    Ok(args)
64}

66/// User specified arguments in the CLI and config files.
67///
68/// Used by [clap::Parser] and [toml::from_str].
69///
70/// Every type **must** be an [Option] in order for the additive nature of
71/// the [configure] function to work.
72///
73/// For CLAP documentation refer to <https://docs.rs/clap/>
74#[derive(Parser, Clone, Debug, PartialEq, Deserialize)]
75#[command(version, about = None, long_about = None)] // allows -v version command
76#[serde(deny_unknown_fields)]
77pub struct Args {
78    /// Absolute or relative path to a config file
79    #[arg(short, long)]
80    config: Option<String>,
81
82    /// Select the granularity of logs
83    #[arg(short, long)]
84    log_level: Option<LogLevelArg>,
85
86    /// Select map security level
87    #[arg(short, long)]
88    map_modification: Option<SecurityLevel>,
89
90    /// Select pin security level
91    #[arg(short, long)]
92    pin_modification: Option<SecurityLevel>,
93
94    /// Is `kill -SIGINT <pid>` allowed to kill userspace process?
95    #[arg(short, long)]
96    sigint: Option<SecurityLevel>,
97
98    /// Are kernel modules allowed to be loaded?
99    #[arg(long)]
100    kmod: Option<SecurityLevel>,
101
102    /// Protection level for modification of daemon service, config, and executable files
103    #[arg(short, long)]
104    daemon_modification: Option<SecurityLevel>,
105
106    /// Whether or not ptrace can be used on seabee. Use this option to run seabee under GDB.
107    /// Note that PTRACE_ATTACH will cause seabee to exit gracefully while PTRACE_SEIZE will allow
108    /// seabee to continue running.
109    #[arg(short('t'), long)]
110    ptrace: Option<SecurityLevel>,
111
112    // A list of LogTypes to exclude from output
113    #[arg(short, long, value_delimiter = ' ', num_args = 0..)]
114    #[serde(default)]
115    exclude: Vec<LogTypeCLI>,
116
117    /// Whether or not digital signature verification for policy updates is enabled. This option is enabled
118    /// by default and disabling it eliminates all security benefits SeaBee provides. Only for debugging.
119    #[arg(short, long)]
120    verify_policy: Option<bool>,
121
122    /// Whether or not digital signature verification for new keys is enabled. This option is
123    /// disabled by default. Enabling allows an administrator to authorize who can create SeaBee
124    /// policies by controlling who can add keys.
125    #[arg(long)]
126    verify_keys: Option<bool>,
127}

129impl Args {
130    /// Update an [Args] struct with values from another [Args]
131    ///
132    /// **Update this when [Args] or Config are updated**
133    pub fn apply(&mut self, other: Args) {
134        if other.log_level.is_some() {
135            self.log_level = other.log_level;
136        }
137        if other.map_modification.is_some() {
138            self.map_modification = other.map_modification;
139        }
140        if other.pin_modification.is_some() {
141            self.pin_modification = other.pin_modification;
142        }
143        if other.sigint.is_some() {
144            self.sigint = other.sigint;
145        }
146        if other.daemon_modification.is_some() {
147            self.daemon_modification = other.daemon_modification;
148        }
149        if other.kmod.is_some() {
150            self.kmod = other.kmod;
151        }
152        if other.ptrace.is_some() {
153            self.ptrace = other.ptrace;
154        }
155        for log in other.exclude {
156            if !self.exclude.contains(&log) {
157                self.exclude.push(log);
158            }
159        }
160        if other.verify_policy.is_some() {
161            self.verify_policy = other.verify_policy;
162        }
163        if other.verify_keys.is_some() {
164            self.verify_keys = other.verify_keys;
165        }
166    }
167}

169impl From<Args> for crate::config::Config {
170    fn from(value: Args) -> Self {
171        // Should not panic since every field must have been initialized with a value from DEFAULT_ARGS
172        let mut config = Self {
173            log_level: value.log_level.unwrap().into(),
174            sigint: value.sigint.unwrap(),
175            kmod: value.kmod.unwrap(),
176            ptrace: value.ptrace.unwrap(),
177            log_filter: {
178                let mut filter = HashSet::new();
179                for log_type in &value.exclude {
180                    for event_type in log_type.to_event_types() {
181                        filter.insert(event_type);
182                    }
183                }
184                filter
185            },
186            verify_policy: value.verify_policy.unwrap(),
187            verify_keys: value.verify_keys.unwrap(),
188            ..Default::default()
189        };
190        config.policy_file.config = value.clone().into();
191        config
192    }
193}

205/// The level of protection desired
206#[derive(
207    Clone,
208    Copy,
209    Debug,
210    Default,
211    serde::Deserialize,
212    serde::Serialize,
213    PartialEq,
214    clap::ValueEnum,
215    zerocopy::IntoBytes,
216    zerocopy::TryFromBytes,
217    zerocopy::Immutable,
218    strum_macros::FromRepr,
219)]
220#[repr(C)]
221pub enum SecurityLevel {
222    /// No protection
223    allow = 1,
224    /// No protection but audit
225    audit = 2,
226    /// Full protection and audit
227    #[default]
228    blocked = 3,
229}
230
231impl std::fmt::Display for SecurityLevel {
232    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
233        match self {
234            Self::allow => write!(f, "allow"),
235            Self::audit => write!(f, "audit"),
236            Self::blocked => write!(f, "blocked"),
237        }
238    }
239}

241/// The least severe log type to output
242#[repr(u32)]
243#[derive(Clone, Copy, Debug, Default, Deserialize, PartialEq, ValueEnum)]
244enum LogLevelArg {
245    off,
246    error,
247    warn,
248    #[default]
249    info,
250    debug,
251    trace,
252}
253
254impl From<LogLevelArg> for LogLevel {
255    fn from(level: LogLevelArg) -> Self {
256        LogLevel::from_repr(level as u32).expect("Bad log level conversion")
257    }

260// Used to generate CLI for log filtering
261#[derive(Clone, Debug, Deserialize, PartialEq, Eq, Hash, ValueEnum)]
262pub enum LogTypeCLI {
263    File,
264    Unmount,
265    Map,
266    Signal,
267    Kmod,
268    Ptrace,
269    BpfWriteUser,
270}
271
272impl LogTypeCLI {
273    /// converts a LogTypeCli to a vec of EventType
274    ///
275    /// Not every EventType is covered by a LogTypeCli
276    fn to_event_types(&self) -> Vec<EventType> {
277        match self {
278            LogTypeCLI::File => vec![
279                EventType::EVENT_TYPE_FILE_OPEN,
280                EventType::EVENT_TYPE_INODE_UNLINK,
281            ],
282            LogTypeCLI::Unmount => vec![EventType::EVENT_TYPE_SB_UMOUNT],
283            LogTypeCLI::Map => vec![EventType::EVENT_TYPE_BPF_MAP],
284            LogTypeCLI::Signal => vec![EventType::EVENT_TYPE_TASK_KILL],
285            LogTypeCLI::Kmod => vec![
286                EventType::EVENT_TYPE_KERNEL_LOAD_DATA,
287                EventType::EVENT_TYPE_KERNEL_READ_FILE,
288                EventType::EVENT_TYPE_KERNEL_MODULE_REQUEST,
289            ],
290            LogTypeCLI::Ptrace => vec![EventType::EVENT_TYPE_PTRACE_ACCESS_CHECK],
291            LogTypeCLI::BpfWriteUser => vec![EventType::EVENT_TYPE_BPF_WRITE_USER],
292        }
293    }
294}
