9impl super::SeaBeePolicy {
10    /// returns a new key id that does not yet exist
11    pub fn get_next_key_id(&mut self) -> Result<u32> {
12        let unused_id = self.next_key_id;
13        match unused_id.checked_add(1) {
14            Some(key_id) => self.next_key_id = key_id,
15            None => {
16                return Err(anyhow!(
17                    "SeaBee ran out of key ids! Restart to refresh ids."
18                ))
19            }
20        }
21        Ok(unused_id)
22    }
23
24    // Verifys a new policy and sets new_policy.key_id
25    // O(n) where n is the number of keys
26    pub fn verify_new_policy(
27        &self,
28        new_policy: &mut super::policy_file::PolicyFile,
29        input_path: &PathBuf,
30        sig_path: &Option<PathBuf>,
31    ) -> Result<()> {
32        // fail if no signature
33        if sig_path.is_none() {
34            return self.interpret_verify_error(anyhow!(
35                "Verification failure for file '{}': no signature given",
36                input_path.display()
37            ));
38        }
39
40        // verify signature, digest comes from policy file
41        if let Some(key) = self.verification_keys.values().find(|key| {
42            self.verify_signature(input_path, sig_path, &None, Some(key.id))
43                .is_ok()
44        }) {
45            new_policy.key_id = Some(key.id)
46        } else {
47            return self.interpret_verify_error(anyhow!(
48                "failed to verify signature for policy at '{}' and sig '{:?}'",
49                input_path.display(),
50                sig_path
51            ));
52        }
53
54        Ok(())
55    }

57    /// wrapper around [crypto::verify_file_signature]
58    ///
59    /// This function will run even when verification is disabled, it will
60    /// return Ok(()) and print a warning on verification failure
61    pub fn verify_signature(
62        &self,
63        obj_path: &PathBuf,
64        sig_path: &Option<PathBuf>,
65        custom_digest: &Option<SeaBeeDigest>,
66        key_id: Option<u32>,
67    ) -> Result<()> {
68        // Error if no signature or key
69        let sig_path = match sig_path {
70            Some(path) => path,
71            None => {
72                return self.interpret_verify_error(anyhow!(
73                    "Verification failure for file '{}': no signature given",
74                    obj_path.display()
75                ))
76            }
77        };
78        let key_id = match key_id {
79            Some(path) => path,
80            None => {
81                return self.interpret_verify_error(anyhow!(
82                    "Verification failure for file '{}': no key id given",
83                    obj_path.display()
84                ))
85            }
86        };
87
88        // Get verification result
89        let digest = crypto::get_digest_algorithm(obj_path, custom_digest)?;
90        let key = match self.verification_keys.get(&key_id) {
91            Some(key) => key,
92            None => {
93                return self.interpret_verify_error(anyhow!(
94                    "Key id '{}' did not exist for verifying object '{}'",
95                    key_id,
96                    obj_path.display()
97                ))
98            }
99        };
100        let verify_result = crypto::verify_file_signature(obj_path, sig_path, digest, key);
101
102        // Check for errors
103        let error = match verify_result {
104            Ok(success) => {
105                if success {
106                    return Ok(());
107                } else {
108                    anyhow!(
109                        "Verification failure\nfile: '{}'\nsig: '{}'\ndigest: '{}'",
110                        obj_path.display(),
111                        sig_path.display(),
112                        digest.type_().short_name()?,
113                    )
114                }
115            }
116            Err(e) => anyhow!(
117                "Error during signature verification\nfile: '{}'\nsig: '{}'\n{e}",
118                obj_path.display(),
119                sig_path.display(),
120            ),
121        };
122
123        self.interpret_verify_error(error)
124    }
125
126    fn interpret_verify_error(&self, error: anyhow::Error) -> Result<()> {
127        if self.verify_policy {
128            return Err(error);
129        } else {
130            warn!("Verification Disabled, got result: \n{error}");
131        }
132        Ok(())
133    }
