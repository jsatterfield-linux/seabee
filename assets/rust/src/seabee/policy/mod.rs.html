19pub const ROOT_KEY_ID: u32 = 0;
20
21#[derive(Debug)]
22pub struct SeaBeePolicy {
23    /// Maps a policy name to a PolicyFile
24    policies: HashMap<String, PolicyFile>,
25    /// SeaBee base policy (the policy that protects SeaBee itself)
26    pub base_policy: PolicyFile,
27    /// The id for the next policy to be assigned. Ensures no two policies have the same id
28    next_policy_id: u32,
29    /// Listens for connections from seabeectl for user interaction
30    listener: Option<UnixListener>,
31    /// Is policy verification enabled?
32    verify_policy: bool,
33    /// Is key verification enabled?
34    verify_keys: bool,
35    /// A list of verification keys including the root key
36    verification_keys: HashMap<u32, crypto::SeaBeeKey>,
37    /// The id for the next key to be assigned.
38    next_key_id: u32,
39}
40
41impl SeaBeePolicy {
42    pub fn init(config: &Config) -> Result<Self> {
43        let mut verification_keys = HashMap::new();
44        let root_key =
45            match crypto::SeaBeeKey::new_key(&PathBuf::from(SEABEE_ROOT_KEY_PATH), ROOT_KEY_ID) {
46                Ok(key) => key,
47                Err(e) => {
48                    return Err(anyhow!(
49                        "Failed to get root key from {SEABEE_ROOT_KEY_PATH}.\n{e}"
50                    ))
51                }
52            };
53        verification_keys.insert(ROOT_KEY_ID, root_key);
54
55        Ok(Self {
56            policies: HashMap::default(),
57            base_policy: config.policy_file.clone(),
58            next_policy_id: policy_file::BASE_POLICY_ID + 1,
59            verification_keys,
60            verify_policy: config.verify_policy,
61            verify_keys: config.verify_keys,
62            listener: None,
63            next_key_id: ROOT_KEY_ID + 1,
64        })
65    }
66
67    /// returns a policy id that has not yet been used
68    /// TODO: inflexible, will need to fit within kernel memory limits
69    pub fn get_unused_policy_id(&mut self) -> Result<u32> {
70        let unused_id = self.next_policy_id;
71        match unused_id.checked_add(1) {
72            Some(next_policy_id) => self.next_policy_id = next_policy_id,
73            None => {
74                return Err(anyhow!(
75                    "SeaBee ran out of policy ids! Restart to refresh ids."
76                ))
77            }
78        }
79        Ok(unused_id)
80    }
81}
