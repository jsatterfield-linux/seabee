22/// Processes configuration information in the order specified below.
23///
24/// 1. Initialize from default arguments.
25/// 1. Apply args from [CONFIG_PATH] (if present on disk).
26/// 1. Apply `--config` CLI arguments (if present).
27/// 1. Apply all other CLI arguments (if present).
28pub fn configure() -> Result<Config> {
29    // Gather the config info using the default trace dispatcher
30    let default = tracing_subscriber::fmt().with_target(false).finish();
31    let config = tracing::subscriber::with_default(default, || -> Result<Config> {
32        let mut base_args = Args::default();
33        init_paths()?;
34        base_args.apply(args_from_file(CONFIG_PATH)?);
35        base_args.apply(args_from_cli()?);
36        Ok(base_args.into())
37    })?;
38
39    configure_logging(config.log_level)?;
40
41    if !config.verify_policy {
42        warn!("Verification of policy updates is disabled. This configuration makes it trivial to defeat all security properties that SeaBee provides.")
43    }
44
45    Ok(config)
46}

48pub fn configure_logging(log_level: LogLevel) -> Result<()> {
49    // Set the global tracing dispatcher with the log level specified in the config
50    let filter =
51        tracing_subscriber::filter::LevelFilter::from_str(&(log_level as usize).to_string())?;
52    LOGGING.call_once(|| {
53        let registry = tracing_subscriber::registry().with(
54            tracing_subscriber::fmt::layer()
55                .with_target(false)
56                .without_time()
57                .with_filter(filter),
58        );
59        match tracing_journald::layer() {
60            Ok(layer) => registry.with(layer).init(),
61            Err(_) => tracing::error!("Unable to connect to journald"),
62        };
63    });
64
65    Ok(())
66}

68pub fn init_paths() -> Result<()> {
69    trace!("create pin path");
70    fs::create_dir_all(PIN_DIR)?;
71    // These folders are used to store key and signature data.
72    // They do not get cleaned up because we need to save policies and keys
73    // through a reboot. We should not error if they exist when we start up.
74    trace!("create seabee directories");
75    utils::create_dir_if_not_exists(SEABEE_DIR)?;
76    utils::create_dir_if_not_exists(POLICY_DIR)?;
77    utils::create_dir_if_not_exists(POL_SIGNATURE_DIR)?;
78    utils::create_dir_if_not_exists(KEY_DIR)?;
79    utils::create_dir_if_not_exists(KEY_SIGNATURE_DIR)?;
80    // These files are part of the SeaBee base policy
81    trace!("create config and service path");
82    utils::open_or_create(CONFIG_PATH)?;
83    utils::open_or_create(SERVICE_PATH)?;
84    trace!("finished creating paths");
85    Ok(())
86}

88/// Parser-complete configuration
89///
90/// Make sure all updates are reflected in [Args] and its functions
91#[derive(Clone, Debug, PartialEq)]
92pub struct Config {
93    pub log_level: LogLevel,
94    pub sigint: SecurityLevel,
95    pub kmod: SecurityLevel,
96    pub ptrace: SecurityLevel,
97    pub policy_file: PolicyFile,
98    pub log_filter: HashSet<EventType>,
99    // will only be true during test cases
100    pub test: bool,
101    // is signature verification for policies enabled?
102    pub verify_policy: bool,
103    // is signature verification for new keys enabled?
104    pub verify_keys: bool,
105}
106
107impl Default for Config {
108    fn default() -> Self {
109        let policy_file = PolicyFile::base().expect("Cannot create base policy");
110
111        // The default config is intended to be a secure, but usable configuration
112        Self {
113            log_level: LogLevel::LOG_LEVEL_INFO,
114            sigint: SecurityLevel::blocked,
115            kmod: SecurityLevel::audit,
116            ptrace: SecurityLevel::blocked,
117            test: false,
118            policy_file,
119            log_filter: HashSet::new(),
120            verify_policy: true,
121            verify_keys: false,
122        }
123    }
124}

126/// sets up ctrl+c handler based on whether or not SIGINT is allowed
127pub fn setup_ctrlc(sigint: SecurityLevel) -> Result<Arc<AtomicBool>> {
128    let running = Arc::new(AtomicBool::new(true));
129    if crate::utils::is_sigint_allowed(sigint) {
130        let r = running.clone();
131        ctrlc::set_handler(move || {
132            r.store(false, Ordering::SeqCst);
133        })?;
134    }
135    Ok(running)
136}
