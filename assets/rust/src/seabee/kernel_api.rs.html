23/// Updates the ebpf "policy_map" with an id and a config. This can
24/// be either an existing id or a new a id.
25pub fn update_kernel_policy_map(
26    maps: &SeaBeeMapHandles,
27    id: u32,
28    config: &policy_file::PolicyConfig,
29) -> Result<()> {
30    let config = config.to_c_policy_config();
31    if let Err(e) = maps
32        .policy
33        .update(id.as_bytes(), config.as_bytes(), libbpf_rs::MapFlags::ANY)
34    {
35        return Err(anyhow!(
36            "Error: update_kernel_policy_map: possibly exceeded max policies: {}\n error: '{}'",
37            constants::SEABEE_MAX_POLICIES,
38            e
39        ));
40    }
41    Ok(())
42}

44/// removes a policy from the kernel based on an id.
45///
46/// This removes the policy from the policy map so there will no longer
47/// be a policy associated with the given policy id. Objects still labeled with this id
48/// will be treated as if they have no policy id.
49pub fn remove_kernel_policy(maps: &SeaBeeMapHandles, policy_id: u32) -> Result<()> {
50    debug!("remove kernel policy: {}", policy_id);
51    maps.policy.delete(policy_id.as_bytes())?;
52
53    // inode and task local storage labels will persist,
54    //   but will not be associated with a policy, so they won't be affected
55    // map protections will be removed when maps are deallocated
56    Ok(())
57}

59// Add a path to the scope corresponding to an id
60pub fn add_path_to_scope(maps: &SeaBeeMapHandles, path: &String, id: u32) -> Result<()> {
61    let key = utils::str_to_bytes(path, PATH_MAX.try_into()?)?;
62
63    if let Err(e) =
64        maps.path_to_pol_id
65            .update(key.as_slice(), id.as_bytes(), libbpf_rs::MapFlags::ANY)
66    {
67        return Err(anyhow!(
68            "Error: add_path_to_scope: possibly exceeded max policy scopes: {}\n error: '{e}'",
69            constants::SEABEE_MAX_POLICY_SCOPES,
70        ));
71    }
72    debug!("added path {} to scope for policy {}", path, id);
73    Ok(())
74}

76/// Label all files for a policy with a given id
77pub fn label_files_from_policy(policy: &PolicyFile, sb_maps: &SeaBeeMapHandles) -> Result<()> {
78    debug!("Labeling files for policy id: {}", policy.id);
79    let mut open_object = MaybeUninit::uninit();
80    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
81
82    // label files
83    for file in &policy.files {
84        if let Err(e) = label_file(&skel.maps.filename_to_policy_id, file, policy.id) {
85            return Err(anyhow!("failed to label file {file}. Error: {e}"));
86        }
87    }
88
89    Ok(())
90}

92/// Label a given file with a given id
93pub fn label_file_with_id(sb_maps: &SeaBeeMapHandles, file: &String, id: u32) -> Result<()> {
94    debug!("Labeling file {file} with id: {id}");
95    let mut open_object = MaybeUninit::uninit();
96    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
97    if let Err(e) = label_file(&skel.maps.filename_to_policy_id, file, id) {
98        return Err(anyhow!("failed to label file {}. Error: {}", file, e));
99    }
100
101    Ok(())

118pub fn label_pins(
119    pins: &Vec<PinnedLink>,
120    policy_id: u32,
121    sb_maps: &SeaBeeMapHandles,
122) -> Result<()> {
123    debug!("Labeling pins...");
124    let mut open_object = MaybeUninit::uninit();
125    let skel = load_label_file_skel(&mut open_object, sb_maps)?;
126
127    for pin in pins {
128        match pin.link.pin_path() {
129            Some(path) => label_file(
130                &skel.maps.filename_to_policy_id,
131                &path.to_string_lossy(),
132                policy_id,
133            )?,
134            None => {
135                return Err(anyhow!("link was not pinned!"));
136            }
137        }
138    }
139    Ok(())
140}

142/// Adds access protections for every map in a skeleton
143pub fn label_maps_for_skel(
144    skel: &dyn Skel,
145    policy_id: u32,
146    sb_maps: &SeaBeeMapHandles,
147) -> Result<()> {
148    debug!("Labeling maps...");
149
150    // Setup
151    let num_maps = skel.object().maps().count();
152    let mut open_object = MaybeUninit::uninit();
153    let label_skel = load_label_maps_skel(&mut open_object, sb_maps, num_maps)?;
154    for map in skel.object().maps() {
155        let key = map.info()?.info.id;
156        label_skel.maps.map_id_to_pol_id.update(
157            key.as_bytes(),
158            policy_id.as_bytes(),
159            libbpf_rs::MapFlags::ANY,
160        )?;
161    }
162
163    // Trigger eBPF via 'bpf_map_get_fd_by_id'
164    // This will trigger for every map on the system, but only the SeaBee maps will be labeled
165    for _map in query::MapInfoIter::default() {
166        continue;
167    }
168
169    Ok(())
170}

172fn label_file(map: &Map, filepath: &str, policy_id: u32) -> Result<()> {
173    // Update map with filename
174    let file_name = Path::new(filepath)
175        .file_name()
176        .context(format!("label_file: file_name() failed on {filepath}"))?;
177    let key = crate::utils::str_to_bytes(
178        file_name
179            .to_str()
180            .context(format!("label_file: to_str() failed on {filepath}"))?,
181        128,
182    )?;
183    map.update(
184        key.as_slice(),
185        policy_id.as_bytes(),
186        libbpf_rs::MapFlags::ANY,
187    )?;
188
189    // trigger 'seabee_label_file' eBPF program in 'label_file.bpf.c' by trying to delete a file
190    match fs::remove_file(filepath) {
191        Ok(_) => Err(anyhow!("{filepath} was not allowed to be deleted")),
192        Err(e) => match e.kind() {
193            ErrorKind::PermissionDenied => Ok(()),
194            _ => Err(anyhow!("unexpected error: {e}")),
195        },
196    }?;
197
198    // Reset map
199    map.delete(key.as_slice())?;
200    Ok(())
201}

203/// Updates task storage for associated pid with the given policy id
204pub fn label_seabee_process(sb: &SeaBee) -> Result<()> {
205    debug!("Labeling seabee process with id: {BASE_POLICY_ID}");
206    let mut open_object = MaybeUninit::uninit();
207    let _skel = load_label_task_skel(&mut open_object, sb, BASE_POLICY_ID)?;
208
209    // trigger ebpf by opening a file
210    fs::File::open("/proc/self/stat")?;
211
212    Ok(())
213}

215fn load_label_file_skel<'a>(
216    open_object: &'a mut MaybeUninit<OpenObject>,
217    sb_maps: &SeaBeeMapHandles,
218) -> Result<label_file::LabelFileSkel<'a>> {
219    let mut open_skel = label_file::LabelFileSkelBuilder::default().open(open_object)?;
220
221    open_skel
222        .maps
223        .inode_storage
224        .reuse_fd(sb_maps.inode_storage.as_fd())?;
225    open_skel
226        .maps
227        .log_ringbuf
228        .reuse_fd(sb_maps.log_ringbuf.as_fd())?;
229    open_skel.maps.bss_data.user_pid = process::id();
230    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
231    let mut skel = open_skel.load()?;
232    skel.attach()?;
233
234    debug!("Label file skel was attached");
235
236    Ok(skel)
237}

239fn load_label_task_skel<'a>(
240    open_object: &'a mut MaybeUninit<OpenObject>,
241    sb: &SeaBee,
242    policy_id: u32,
243) -> Result<label_task::LabelTaskSkel<'a>> {
244    let mut open_skel = label_task::LabelTaskSkelBuilder::default().open(open_object)?;
245    open_skel
246        .maps
247        .task_storage
248        .reuse_fd(sb.maps.task_storage.as_fd())?;
249    open_skel
250        .maps
251        .log_ringbuf
252        .reuse_fd(sb.maps.log_ringbuf.as_fd())?;
253    open_skel.maps.bss_data.user_pid = process::id();
254    open_skel.maps.bss_data.policy_id = policy_id;
255    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
256    let mut skel = open_skel.load()?;
257    skel.attach()?;
258
259    debug!("Label task skel was attached");
260
261    Ok(skel)
262}

264fn load_label_maps_skel<'a>(
265    open_object: &'a mut MaybeUninit<OpenObject>,
266    sb_maps: &SeaBeeMapHandles,
267    num_maps: usize,
268) -> Result<label_maps::LabelMapsSkel<'a>> {
269    let mut open_skel = label_maps::LabelMapsSkelBuilder::default().open(open_object)?;
270    open_skel.maps.bss_data.log_level = *bpf::logging::LOG_LEVEL.get().unwrap() as u32;
271    open_skel
272        .maps
273        .map_to_pol_id
274        .reuse_fd(sb_maps.map_to_pol_id.as_fd())?;
275    open_skel
276        .maps
277        .log_ringbuf
278        .reuse_fd(sb_maps.log_ringbuf.as_fd())?;
279    open_skel
280        .maps
281        .map_id_to_pol_id
282        .set_max_entries(num_maps.try_into()?)?;
283    let mut skel = open_skel.load()?;
284    skel.attach()?;
285
286    debug!("Label maps skel was attached");
287
288    Ok(skel)
