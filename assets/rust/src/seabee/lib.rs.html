54/// Load all of the eBPF programs in the desired configuration.
55///
56/// 1. The eBPF code was converted to multiple [libbpf_rs::skel] during compilation.
57/// 1. First the 'enforce skeleton' containing eBPF LSM programs is loaded.
58/// 1. Next, 'policy skeleton' is laoded containing programs for updating policy.
59/// 1. Each program is attached to its hook point and pinned to the eBPF filesystem.
60pub fn seabee_init(config: Config, open_obj: &mut MaybeUninit<OpenObject>) -> Result<SeaBee> {
61    // set the static LOG_LEVEL so that it can be used by various skeletons
62    LOG_LEVEL.get_or_init(|| config.log_level);
63    LOG_FILTER.get_or_init(|| config.log_filter.clone());
64
65    print_debug_info(&config);
66
67    let policy = SeaBeePolicy::init(&config)?;
68    let mut sb = match enforce::load_ebpf(config.clone(), policy, open_obj) {
69        Ok(sb) => sb,
70        // failsafe in case the SeaBee structure is not created
71        // before an error occurred
72        Err(e) => {
73            crate::unload_ebpf::cleanup_return();
74            return Err(e);
75        }
76    };
77    sb.policy.boot_time_key_load()?;
78    sb.policy.boot_time_policy_load(&sb.maps)?;
79    sb.policy.init_listener()?;
80    // Label seabeectl to enable runtime updates
81    kernel_api::add_path_to_scope(
82        &sb.maps,
83        &constants::SEABEECTL_EXE.to_string(),
84        policy::policy_file::BASE_POLICY_ID,
85    )?;
86    debug!("exit SeaBee init");
87    Ok(sb)
88}

90pub fn main_loop(
91    mut seabee: SeaBee,
92    ctrlc_handler: Arc<std::sync::atomic::AtomicBool>,
93) -> Result<()> {
94    debug!("Starting main loop...");
95
96    let log_rb = bpf::logging::setup_logger(&seabee.maps.log_ringbuf)?;
97
98    let timeout = std::time::Duration::from_millis(10);
99    while ctrlc_handler.load(std::sync::atomic::Ordering::SeqCst) {
100        // sync policy
101        policy::runtime_update::runtime_policy_update(&mut seabee)?;
102        // check for logs
103        if let Err(e) = log_rb.poll(timeout) {
104            // sometimes SIGINT will interrupt poll() which is okay
105            if e.kind() != libbpf_rs::ErrorKind::Interrupted {
106                return Err(e.into());
107            }
108        }
109    }
110    info!("Exiting main loop");
111    Ok(())
112}

114/// Prints welcome message
115fn print_debug_info(config: &Config) {
116    info!("Welcome to the SeaBee userspace!");
117    if utils::is_sigint_allowed(config.sigint) {
118        info!("Kill the userspace with the Ctrl+C shortcut");
119    }
120    info!("See daemon output with `journalctl -u seabee -f`");
121    info!("If cannot remove seabee another way, you will have to 'sudo reboot'");
122    debug!("{:#?}", config);
123}
