104static __always_inline int label_map_with_id(struct bpf_map *map, u32 policy_id)
105{
106 // unsure if map can be null, but this made the verifier happy
107 if (!map) {
108 return ALLOW;
109 }
110
111 struct bpf_map_data map_data;
112 map_data.policy_id = policy_id;
113 BPF_CORE_READ_STR_INTO(&map_data.name, map, name);
114
115 long err = bpf_map_update_elem(&map_to_pol_id, &map, &map_data, BPF_ANY);
116 u64 data[3] = { (u64)map_data.name, (u64)BPF_CORE_READ(map, id),
117 (u64)policy_id };
118 if (err < 0) {
119 log_generic_msg(LOG_LEVEL_ERROR, LOG_REASON_ERROR,
120 "Error: update elem failed map %s(%d) for policy %d",
121 data, sizeof(data));
122 } else {
123 log_generic_msg(LOG_LEVEL_TRACE, LOG_REASON_DEBUG,
124 "label map %s(%d) as %d", data, sizeof(data));
125 }
126 return ALLOW;
127}

143static __always_inline int unlabel_map(struct bpf_map *map)
144{
145 // unsure if map can be null, but this made the verifier happy
146 if (!map) {
147 return ALLOW;
148 }
149
150 // exit early if map is not tracked
151 u32 *policy_id = bpf_map_lookup_elem(&map_to_pol_id, &map);
152 if (!policy_id) {
153 return ALLOW;
154 }
155
156 // delete map
157 s32 err = bpf_map_delete_elem(&map_to_pol_id, &map);
158 char map_name[BPF_MAP_NAME_LEN];
159 BPF_CORE_READ_STR_INTO(&map_name, map, name);
160 u64 data[] = { (u64)map_name, (u64)BPF_CORE_READ(map, id) };
161 if (err < 0) {
162 log_generic_msg(LOG_LEVEL_ERROR, LOG_REASON_ERROR,
163 "Error: failed to unlabel map: %s(%d)", data,
164 sizeof(data));
165 } else {
166 log_generic_msg(LOG_LEVEL_TRACE, LOG_REASON_DEBUG, "unlabel map %s(%d)",
167 data, sizeof(data));
168 }
169 return ALLOW;
170}
