41static inline void *log_buf(enum LogLevel level, enum LogReason reason,
42 enum EventType type, size_t size,
43 unsigned long pol_id)
44{
45 if (level > log_level)
46 return NULL;
47
48 void *log = bpf_ringbuf_reserve(&log_ringbuf, size, 0);
49 if (!log) {
50 bpf_printk("Unable to reserve from ringbuf of size %lu", size);
51 return NULL;
52 }
53
54 struct log_hdr *hdr = (struct log_hdr *)log;
55 hdr->level = level;
56 hdr->type = type;
57 hdr->reason = reason;
58 // kernel tgid == user space process id
59 // kernel pid == user space thread id
60 u64 pid_tgid = bpf_get_current_pid_tgid();
61 hdr->pid = pid_tgid >> 32;
62 hdr->tid = pid_tgid & 0xFFFFFFFF;
63 hdr->uid = bpf_get_current_uid_gid() & 0xFFFFFFFF;
64 hdr->pol_id = pol_id;
65 bpf_get_current_comm(hdr->comm, sizeof(hdr->comm));
66 return log;
67}
